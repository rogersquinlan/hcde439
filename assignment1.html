Here is my arduino code.
void setup() { // initialize digital pin 13 as an output. pinMode(13, OUTPUT); // initialize digital pin 12 as an output. pinMode(12, OUTPUT); // initialize digital pin 9 as an output. pinMode(9, OUTPUT); } // the loop function runs over and over again forever void loop() { digitalWrite(13, HIGH); // turn 13 LED on (HIGH is the voltage level) delay(1000); // wait for a second digitalWrite(13, LOW); // turn 13 LED off by making the voltage LOW delay(1000); // wait for a second digitalWrite(12, HIGH); // turn 12 LED on (HIGH is the voltage level) delay(1000); // wait for a second digitalWrite(12, LOW); // turn 12 LED off by making the voltage LOW delay(1000); // wait for a second digitalWrite(9, HIGH); // turn 9 LED on (HIGH is the voltage level) delay(1000); // wait for a second digitalWrite(9, LOW); // turn 9 LED off by making the voltage LOW delay(1000); // wait for a second }
